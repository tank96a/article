from ctypes import c_uint64
crctable = []
untable=[]
for i in range(256):
    t = c_uint64(i)
    for _i in range(8):
        b = t.value & 0x1
        t.value = t.value >> 0x1
        if b:
            t.value ^= 0xc96c5795d7870f42
    untable.append(t.value >> 56)
    crctable.append(t.value)

def find_index(crc):
    return untable.index(crc>>56)

def crc(c,crc_):
    index=(crc_^c)&0xff
    crc_=crc_>>8
    crc_=crc_^crctable[index]
    return crc_

def uncrc(c,crc_):
    t=find_index(crc_)
    crc_^=crctable[t]
    crc_=(crc_<<8)&0xffffffffffffffff
    crc_=crc_+(t^c)
    return crc_
